Quick Start Example

JavaScript
Python
Go
// Install the SDK
npm install bitcoin-starknet-bridge-sdk

// Import and initialize
import { BitcoinStarknetBridge } from 'bitcoin-starknet-bridge-sdk';

const bridge = new BitcoinStarknetBridge({
  apiKey: process.env.BRIDGE_API_KEY,
  network: 'mainnet'
});

// Bridge Bitcoin to Starknet
async function transferBitcoin() {
  try {
    const result = await bridge.deposit({
      amount: 0.1,
      starknetAddress: '0x1234...abcd',
      bitcoinAddress: 'bc1qar0srrr7xfkvy5l643lydnw9re59gtzzwf5mdq'
    });
    console.log('Transaction ID:', result.transactionId);
  } catch (error) {
    console.error('Transfer failed:', error.message);
  }
}
