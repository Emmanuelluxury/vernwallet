API Example Code


// Initialize Bitcoin-StarkNet Bridge SDK
const { BridgeSDK } = require('btc-starknet-bridge-sdk');

// Create bridge instance
const bridge = new BridgeSDK({
  network: 'mainnet',
  apiKey: process.env.BRIDGE_API_KEY,
});

// Transfer BTC to StarkNet
async function transferBTCtoStarkNet(amount, recipient) {
  try {
    const transaction = await bridge.transfer({
      from: 'bitcoin',
      to: 'starknet',
      amount: amount,
      recipient: recipient,
      currency: 'BTC'
    });
    
    console.log('Transaction initiated:', transaction.id);
    console.log('BTC deposit address:', transaction.depositAddress);
    
    return transaction;
  } catch (error) {
    console.error('Transfer failed:', error.message);
  }
}

// Execute transfer
transferBTCtoStarkNet(0.1, '0x1234...abcd');

// SPDX-License-Identifier: MIT
pragma solidity ^0.8.0;

import "@openzeppelin/contracts/security/ReentrancyGuard.sol";
import "@openzeppelin/contracts/access/Ownable.sol";

contract BitcoinStarkNetBridge is ReentrancyGuard, Ownable {
    // Bridge state variables
    mapping(bytes32 => bool) public processedTransactions;
    uint256 public minTransferAmount;
    uint256 public maxTransferAmount;
    uint256 public totalTransferred;
    
    // Events
    event BridgeInitialized(uint256 minAmount, uint256 maxAmount);
    event BTCReceived(address indexed recipient, uint256 amount, bytes32 btcTxHash);
    event BTCSent(address indexed sender, uint256 amount, string btcAddress);
    
    constructor(uint256 _minTransferAmount, uint256 _maxTransferAmount) {
        minTransferAmount = _minTransferAmount;
        maxTransferAmount = _maxTransferAmount;
        emit BridgeInitialized(_minTransferAmount, _maxTransferAmount);
    }
    
    // Lock BTC for StarkNet transfer
    function lockBTC(uint256 amount, string calldata btcAddress) 
        external 
        payable 
        nonReentrant 
    {
        require(amount >= minTransferAmount, "Amount below minimum");
        require(amount <= maxTransferAmount, "Amount exceeds maximum");
        require(msg.value == amount, "Incorrect BTC amount");
        
        // Process the lock
        totalTransferred += amount;
        emit BTCSent(msg.sender, amount, btcAddress);
    }
    
    // Release BTC from StarkNet
    function releaseBTC(
        address recipient, 
        uint256 amount, 
        bytes32 btcTxHash,
        bytes calldata signature
    ) external nonReentrant onlyOwner {
        require(!processedTransactions[btcTxHash], "Transaction already processed");
        require(amount > 0, "Invalid amount");
        
        // Verify the transaction signature
        bytes32 messageHash = keccak256(abi.encodePacked(recipient, amount, btcTxHash));
        address signer = recoverSigner(messageHash, signature);
        require(signer == owner(), "Invalid signature");
        
        // Mark transaction as processed
        processedTransactions[btcTxHash] = true;
        
        // Transfer funds to recipient
        payable(recipient).transfer(amount);
        emit BTCReceived(recipient, amount, btcTxHash);
    }
    
    // Helper function to recover signer
    function recoverSigner(bytes32 messageHash, bytes memory signature) 
        internal 
        pure 
        returns (address) 
    {
        (bytes32 r, bytes32 s, uint8 v) = splitSignature(signature);
        return ecrecover(messageHash, v, r, s);
    }
    
    // Helper function to split signature
    function splitSignature(bytes memory signature)
        internal
        pure
        returns (bytes32 r, bytes32 s, uint8 v)
    {
        require(signature.length == 65, "Invalid signature length");
        assembly {
            r := mload(add(signature, 32))
            s := mload(add(signature, 64))
            v := byte(0, mload(add(signature, 96)))
        }
    }
}
